77 void rcInit(void) {
78  TIMSK0 |= (1 << TOIE0); // Enable Overflow Interrupt
79  TCNT0 = RC_TIMER_RELOAD; // Overflow after 3,68ms
80  TCCR0B |= (1 << CS02); // Prescaler 256

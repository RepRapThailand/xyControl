139 void flightTask(void) {
140  static time_t last = 100; // Don't begin immediately
141  if ((getSystemTime() - last) >= QUADDELAY) {
142  last = getSystemTime();
143  Error e = orientationTask();
144  REPORTERROR(e);
145  if (state & 0x02) {
146  pidTask();
147  } else {
148  o_output[0] = o_output[1] = 0;
149  }
150  setTask();
151  motorTask();
152 
153  uint32_t diff = getSystemTime() - last;
154  if (++sumFlightCount >= QUADFREQ) {
155  sumFlightCount = 1;
156  sumFlightTask = diff;
157  } else {
158  sumFlightTask += diff;
159  }
160  }
161 }

163 void statusTask(void) {
164  static time_t last = 100; // Don't begin immediately
165  static uint32_t lastDuration = 0;
166  if (((getSystemTime() - last) >= STATUSDELAY) && (!(state & STATE_OUTPUT))) {
167  last = getSystemTime();
168  printf("q%li %li\n", sumFlightTask / sumFlightCount, lastDuration);
169  printf("r%.2f %.2f\n", o_pids[0].intMin, o_pids[0].intMax);
170  printf("s%.2f %.2f\n", o_pids[0].outMin, o_pids[0].outMax);
171  printf("t%.3f %.3f %.3f\n", o_pids[0].kp, o_pids[0].ki, o_pids[0].kd);
172  printf("u%.2f %.2f\n", o_output[PITCH], o_output[ROLL]);
173  printf("v%i %i %i %i\n", motorSpeed[0], motorSpeed[1], motorSpeed[2], motorSpeed[3]);
174  printf("w%.2f\n", orientation.pitch);
175  printf("x%.2f\n", orientation.roll);
176  printf("y%.2f\n", orientation.yaw);
177  printf("z%.2f\n", getVoltage());
178  lastDuration = getSystemTime() - last;
179  }
